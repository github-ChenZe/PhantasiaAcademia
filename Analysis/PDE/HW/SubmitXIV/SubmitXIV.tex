\newprob{5.11 (1)}%
须求解如下问题的基本解:
\begin{align*}
    & U_{tt} = a^2 U_{xx} - b^2 U,\quad U_{t=0} = 0,\quad U_t\vert_{t=0} = \delta\pare{x}, \\
    & \xLongrightarrow{\hat U = F_x\brac{U}} \hat U_{tt} = -a^2 p^2 \hat U - b^2 \hat U,\quad \hat U_{t=0} = 0,\quad \hat U_t\vert_{t=0} = 1, \\

\end{align*}

\newprobheader{(2)}%
须求解如下问题的基本解:
\begin{align*}
    & U_{t} = a^2\laplacian U_{xx} - U,\quad U_{t=0} = \delta\pare{x,y}, \\
    & \xLongrightarrow{\hat U = F_x\brac{U}} \hat U_{t} = -a^2 p^2 \hat U - \hat U,\quad \hat U_{t=0} = 1, \\
    & \Rightarrow \hat U = e^{-\pare{a^2p^2 + 1}t}. \\
    & \Rightarrow U = \frac{e^{-t}}{\pare{2\pi}^2}\int_{-\infty}^\infty e^{-a^2 tu^2} e^{iux}\,\rd{x}\,\int_{-\infty}^\infty e^{-a^2 tv^2} e^{ivx}\,\rd{x} = \boxed{\rec{4\pi a^2 t}e^{-t-\frac{x^2+y^2}{4a^2t}}.} \\
    & \Rightarrow \boxed{u\pare{t,x,y} = U\pare{t,x,y}*\varphi\pare{x,y} + \int_0^t U\pare{t-\tau,x,y}*f\pare{\tau,x,y}\,\rd{\tau}.}
\end{align*}

\newprobheader{(4)}%
二维波动方程有基本解
\[ U\pare{t,x,y} = \frac{H\pare{at-r}}{2\pi a \sqrt{\pare{at}^2 - r^2}},\quad r = \sqrt{x^2 + y^2}. \]
从而
\begin{align*}
    u &= \+DtD{}\pare{U\pare{t,x,y}*\varphi} = \rec{2\pi a}\+DtD{} \iint_{D_{at}} \rd{\xi}\,\rd{\eta}\, \frac{\xi^2\pare{\xi + \eta}}{\sqrt{\pare{at}^2 - \pare{\xi - x}^2 - \pare{\eta - y}^2}} \\
    &=\rec{2\pi a}\+DtD{} \int_0^{2\pi}\rd{\theta}\,\int_0^{at}\rd{r}\, r\cdot \frac{\pare{x+r\cos\theta}^2\pare{x+y+r\cos\theta + r\sin\theta}}{\sqrt{\pare{at}^2 - r^2}} \\
    &= \rec{2\pi a}\+DtD{} \int_0^{at}\rd{r} \,\int_0^{2\pi}\rd{\theta}\, \frac{x\pare{x+y} + 2xr^2\cos^2\theta + \pare{x+y}r^2\cos^2\theta}{\sqrt{\pare{at}^2 - r^2}}\cdot r \\
    &= \rec{a}\+DtD{}\int_0^{at} \frac{x^2\pare{x+y} + xr^2 + \pare{x+y}r^2/2}{\sqrt{\pare{at}^2 - r^2}}\cdot r \\
    &= \rec{2a}\+DtD{}\int_0^{\pare{at}^2} \frac{x^2\pare{x+y} + xu + \pare{x+y}u/2}{\sqrt{\pare{at}^2 - u}} \\
    &= \rec{2a} \+DtD{}\brac{x^2\pare{x+y}\cdot 2at + \pare{x+\frac{x+y}{2}}\cdot \frac{4}{3}a^3 t^3} \\
    & \boxed{x^2\pare{x+y} + \pare{3x+y}a^2t^2.}
\end{align*}
