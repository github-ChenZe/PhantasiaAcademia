\newprob{5.9 (1)}%

\[ G\pare{x,y;x_0,y_0}  = \rec{2\pi}\ln \abs{\frac{z^2 - z_0^2}{z^2 - \conj{z}_0^2}} = \boxed{\rec{4\pi}\ln \frac{\brac{\pare{x-x_0}^2 + \pare{y-y_0}^2}\brac{\pare{x+x_0}^2 + \pare{y+y_0}^2}}{\brac{\pare{x-x_0}^2 + \pare{y+y_0}^2}\brac{\pare{x+x_0}^2 + \pare{y-y_0}^2}}.} \]

\newprob{5.13}%
$\bullet\quad$$u_t = Lu + f\pare{t,x}$:
\begin{align*}
    u_t &= \int_{x_1}^{x_2}\varphi\pare{\xi}\+DtD{}G\pare{t,x;0,\xi}\,\rd{\xi} + \int_{x_1}^{x_2} f\pare{t,\xi} G\pare{t,x;t,\xi}\,\rd{\xi}\\ &\phantom{=\ }+ \int_0^t\,\rd{\tau} \int_{x_1}^{x_2} f\pare{\tau,\xi}\+DtD{}G\pare{t,x;\tau,\xi}\,\rd{\xi} \\
    &= \int_{x_1}^{x_2}\varphi\pare{\xi}LG\pare{t,x;0,\xi}\,\rd{\xi} + \int_0^t\,\rd{\tau} \int_{x_1}^{x_2} f\pare{\tau,\xi}LG\pare{t,x;\tau,\xi}\,\rd{\xi}\\ &\phantom{=\ } + \int_{x_1}^{x_2} f\pare{t,\xi} \delta\pare{x-\xi}\,\rd{\xi} \\
    &= L\curb{ \int_{x_1}^{x_2}\varphi\pare{\xi}G\pare{t,x;0,\xi}\,\rd{\xi} + \int_0^t\,\rd{\tau} \int_{x_1}^{x_2} f\pare{\tau,\xi}G\pare{t,x;\tau,\xi}\,\rd{\xi} } + f\pare{t,\xi} \\
    &= Lu + f\pare{t,\xi}.
\end{align*}
$\bullet\quad$$\pare{\alpha_1 u_x - \beta_1 u}\vert_{x=x_1} = 0$, $\pare{\alpha_2 u_x + \beta_2 u}\vert_{x=x_2} = 0$:
\begin{align*}
    \pare{\alpha_1 u_x - \beta_1 u}\vert_{x=x_1} &= \int_{x_1}^{x_2}\varphi\pare{\xi}\brac{\alpha_1 \+DxD{}G\pare{t,x;0,\xi} - \beta_1 G\pare{t,x;0,\xi}}_{x=x_1}\,\rd{\xi} \\
    &\phantom{=\ } \int_0^t\,\rd{\tau} \int_{x_1}^{x_2}f\pare{\tau,\xi}\brac{\alpha_1 \+DxD{}G\pare{t,x;0,\xi} - \beta_1 G\pare{t,x;0,\xi}}_{x=x_1}\,\rd{\xi} \\
    &= 0+0 = 0. \\
    \pare{\alpha_2 u_x + \beta_2 u}\vert_{x=x_2} &= \int_{x_1}^{x_2}\varphi\pare{\xi}\brac{\alpha_2 \+DxD{}G\pare{t,x;0,\xi} + \beta_2 G\pare{t,x;0,\xi}}_{x=x_2}\,\rd{\xi} \\
    &\phantom{=\ } \int_0^t\,\rd{\tau} \int_{x_1}^{x_2}f\pare{\tau,\xi}\brac{\alpha_2 \+DxD{}G\pare{t,x;0,\xi} + \beta_2 G\pare{t,x;0,\xi}}_{x=x_2}\,\rd{\xi} \\
    &= 0+0 = 0.
\end{align*}
$\bullet\quad$$u\vert_{t=0} = \varphi\pare{x}$:
\[ u\pare{0,x} = \int_{x_1}^{x_2} \varphi\pare{\xi}G\pare{0,x;0,\xi}\,\rd{\xi} = \int_{x_1}^{x_2} \varphi\pare{\xi}\delta\pare{x-\xi}\,\rd{\xi} = \varphi\pare{x}. \]
