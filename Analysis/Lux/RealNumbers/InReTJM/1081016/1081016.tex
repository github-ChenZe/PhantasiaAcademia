    \rd{v} = \frac{u \rd{u}}{v} &\Rightarrow -\int u^4\,\rd{v} \\
    &\Rightarrow -\int \pare{v^2-1}^2\,\rd{v} \\
    &\Rightarrow -v + \frac{2 v^3}{3} - \frac{v^5}{5}.
