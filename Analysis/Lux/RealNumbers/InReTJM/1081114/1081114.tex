\begin{align*}
    \lim_{x\rightarrow 0} \frac{1-x^2 - e^{-x^2}}{x \sin^3 2x} &= \lim_{x\rightarrow 0} \frac{1-x^2 - \pare{1 - x^2 + x^4/2 + \cdots}}{x\pare{2x+\cdots}^3} \\
    &= \lim_{x\rightarrow 0} \frac{-x^4/2}{8x^4} \\
    &= -\rec{16}.
\end{align*}
